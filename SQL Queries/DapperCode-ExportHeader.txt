Things you need to do in Visual Studio 

1. Create a Blazor Server project (https://youtu.be/XsCfaJCPnbI)

2. Add Dapper and Microsoft.Data.SqlClient to the Solution (https://www.youtube.com/watch?v=VeCjL9JwPBM)

3. Create your Database (https://youtu.be/RkzRci8SNCw)

4. Just once per database (not for each table in the database), add the correct connectionstring to appsettings.json (see https://www.youtube.com/watch?v=5XyPWMayg2Q)

"ConnectionStrings":{
     "SqlDBcontext" : "Your connection string here"
}

5. Just once per database (not each table) in  your Data folder create a file named SqlConnectionConfiguration.cs and
   add to it the following code.

---------------------------- /DATA/SqlConnectionConfiguration.cs
namespace Export.Data

--------------  Stored Proc for SELECT (LIST, just first six fields but you can change in final code.)
CREATE PROCEDURE spExportHeader_List
--No parameters required.
AS
BEGIN
--SQL for Select stored procedure.
SELECT TOP 30 Headerid, InvoiceNo, InvoiceDate, SerialNo, ConsignatarioID, AduanasID FROM ExportHeader ORDER BY Headerid DESC
END

--------------  Stored Proc for SELECT (one)
CREATE PROCEDURE spExportHeader_GetOne
-- Needs one parameter for primary key
@Headerid int
AS 
BEGIN
-- SQL Select for one table row
SELECT Headerid, InvoiceNo, InvoiceDate, SerialNo, ConsignatarioID, AduanasID, CountryID, CargadorID, BoardingDate, ExchangeRate, Description, DocTypeID, DuaSimplificada, Complementaria, IncotermID, Closed, HandlerID, SupervisorID, AnulledDate, PrimaImportada FROM ExportHeader WHERE Headerid= @Headerid
END


--------------  Stored Proc for SELECT LIKE
CREATE PROCEDURE spExportHeader_Search
-- One parameter required to identify row to delete.
@Param varchar(128)
AS
BEGIN
-- SQL for search looking for embedded content. 
SELECT InvoiceNo, InvoiceDate, SerialNo, ConsignatarioID, AduanasID FROM ExportHeader WHERE CAST(InvoiceNo AS varchar(20)) LIKE '%' + @param + '%' OR CONVERT(varchar(12),InvoiceDate,101) LIKE + '%' + @param + '%' OR SerialNo LIKE '%' + @param + '%' OR CAST(ConsignatarioID AS varchar(20)) LIKE '%' + @param + '%' OR CAST(AduanasID AS varchar(20)) LIKE '%' + @param + '%' OR CAST(CountryID AS varchar(20)) LIKE '%' + @param + '%' OR CAST(CargadorID AS varchar(20)) LIKE '%' + @param + '%' OR CONVERT(varchar(12),BoardingDate,101) LIKE + '%' + @param + '%' OR CAST(ExchangeRate AS varchar(20)) LIKE '%' + @param + '%' OR Description LIKE '%' + @param + '%' OR CAST(DocTypeID AS varchar(20)) LIKE '%' + @param + '%' OR CAST(DuaSimplificada AS varchar(20)) LIKE '%' + @param + '%' OR Complementaria LIKE '%' + @param + '%' OR CAST(IncotermID AS varchar(20)) LIKE '%' + @param + '%' OR CAST(HandlerID AS varchar(20)) LIKE '%' + @param + '%' OR CAST(SupervisorID AS varchar(20)) LIKE '%' + @param + '%' OR CAST(AnulledDate AS varchar(20)) LIKE '%' + @param + '%' 
END

    // Set default page title and button text
    public string pagetitle = "Add a ExportHeader";
    public string buttontext = "Add";

    //Executes on page open, set defaults on page.
    protected override async Task OnInitializedAsync()
    {
        // ============ If the passed-in Headerid is zero, assume new ExportHeader.
        if (Headerid == 0)
        {
            // Use this block to set for defaults (optional)
        }
        else
        {
            exportheader = await ExportHeaderService.ExportHeader_GetOne(Headerid);
            // Change page title and button text since this is an edit.
            pagetitle = "Edit ExportHeader";
            buttontext = "Update";
        }
    }
    // Executes OnValidSubmit of EditForm above.
    protected async Task ExportHeaderSave()
    {
        if (exportheader.Headerid == 0)
        {
            // Insert if Headerid is zero.
            await ExportHeaderService.ExportHeaderInsert(exportheader);
        }
        else
        {
            // Update if Headerid not 0
            await ExportHeaderService.ExportHeaderUpdate(exportheader);
        }
        NavigationManager.NavigateTo("/exportheaderlist");
    }
    //Executes if user clicks the Cancel button.
    protected void Cancel()
    {
        NavigationManager.NavigateTo("/exportheaderlist");
    }
}
